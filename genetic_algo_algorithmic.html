Genetic Algorithm for Feature Selection

Algorithm 1: Mutate Individual
procedure Mutate(individual)
1: mutated ← copy(individual)
2: selected ← {i : mutated[i] = 1}
3: unselected ← {i : mutated[i] = 0}
4: if selected ≠ ∅ and unselected ≠ ∅ then
5: swap_out ← random_choice(selected)
6: swap_in ← random_choice(unselected)
7: mutated[swap_out] ← 0
8: mutated[swap_in] ← 1
9: end if
10: return mutated
end procedure
Algorithm 2: Crossover Two Parents
procedure Crossover(parent₁, parent₂)
1: num_features ← Σparent₁
2: ones_p1 ← {i : parent₁[i] = 1}
3: ones_p2 ← {i : parent₂[i] = 1}
4: all_ones ← ones_p1 ∪ ones_p2
5: child₁ ← [0, 0, …, 0] ▷ Length = |parent₁|
6: child₂ ← [0, 0, …, 0]
7: selected₁ ← random_sample(all_ones, num_features)
8: for pos ∈ selected₁ do
9: child₁[pos] ← 1
10: end for
11: selected₂ ← random_sample(all_ones, num_features)
12: for pos ∈ selected₂ do
13: child₂[pos] ← 1
14: end for
15: return child₁, child₂
end procedure
Algorithm 3: Keep Elite Individuals
procedure KeepElite(new_pop, fitness, population, keep_top)
1: rank_indices ← argsort(fitness, descending)
2: elite_indices ← rank_indices[0:keep_top]
3: for idx ∈ elite_indices do
4: new_pop.append(copy(population[idx]))
5: end for
6: return new_pop, rank_indices
end procedure
Algorithm 4: Update Population
procedure UpdatePopulation(population, fitness, keep_top, pop_size)
1: new_pop ← [ ]
2: new_pop, rank_indices ← KeepElite(new_pop, fitness, population, keep_top)
3: while |new_pop| < pop_size do
4: parent₁ ← population[rank_indices[0]] ▷ Best individual
5: parent₂ ← population[rank_indices[-1]] ▷ Worst individual
6: child₁, child₂ ← Crossover(parent₁, parent₂)
7: child₁ ← Mutate(child₁)
8: child₂ ← Mutate(child₂)
9: new_pop.extend([child₁, child₂])
10: end while
11: population ← new_pop[0:pop_size]
12: return population
end procedure
Algorithm 5: Create Initial Population
procedure CreatePopulation(search_space, pop_size, num_features)
1: population ← [ ]
2: for i = 1 to pop_size do
3: individual ← [0, 0, …, 0] ▷ Length = |search_space|
4: available ← {0, 1, …, |search_space| − 1}
5: selected ← random_sample(available, num_features)
6: for pos ∈ selected do
7: individual[pos] ← 1
8: end for
9: population.append(individual)
10: end for
11: return population
end procedure
Algorithm 6: Main Genetic Algorithm
procedure GeneticAlgorithm(simulate, search_space, create_entry, underlying, holidays)
1: population ← CreatePopulation(search_space, POP_SIZE, NUM_FEATURES)
2: best_fitness ← −999
3: best_fitness_overall ← −999
4: best_individual_overall ← None
5: for gen = 1 to NUM_GENERATIONS do
6: fitness_scores ← [ ]
7: for individual ∈ population do
8: selected_cols ← {search_space[i] : individual[i] = 1}
9: entry_func ← CreateEntry(selected_cols, underlying, holidays)
10: fitness ← Simulate(entry_func)
11: fitness_scores.append(fitness)
12: if fitness > best_fitness then
13: best_fitness ← fitness
14: end if
15: if fitness > best_fitness_overall then
16: best_fitness_overall ← fitness
17: best_individual_overall ← copy(individual)
18: end if
19: end for
20: avg_fitness ← mean(fitness_scores)
21: print avg_fitness, best_fitness
22: if gen < NUM_GENERATIONS then
23: population ← UpdatePopulation(population, fitness_scores, KEEP_TOP, POP_SIZE)
24: end if
25: end for
26: selected_cols ← {search_space[i] : best_individual_overall[i] = 1}
27: return selected_cols
end procedure
