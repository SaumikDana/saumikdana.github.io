# Sample Solution: Top N with Percentiles
def analyze_employees(df):
    # Add salary percentile within department
    df['salary_percentile'] = df.groupby('department')['salary'].rank(pct=True)
    
    # Get top 3 per department
    top_employees = df.groupby('department').apply(
        lambda x: x.nlargest(3, 'salary')
    ).reset_index(drop=True)
    
    return top_employees[['name', 'department', 'salary', 'salary_percentile']]

# Sample Solution: Month-over-Month Growth
def calculate_mom_growth(df):
    df['date'] = pd.to_datetime(df['date'])
    df['month'] = df['date'].dt.to_period('M')
    
    monthly_sales = df.groupby(['category', 'month'])['sales'].sum()
    growth_rate = monthly_sales.groupby('category').pct_change()
    
    return growth_rate.reset_index()
