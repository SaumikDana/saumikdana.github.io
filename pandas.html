# v1: Basic Implementation
def filter_v1(options, dict_options):
    all_syms = set()
    for dataframe in dict_options.values():
        all_syms.update(dataframe['sym'].tolist())
    return options[options['symbol'].isin(all_syms)]

# v4: Optimized Implementation (60-80% memory reduction)
def filter_v4(options, dict_options):
    # Vectorized symbol collection
    all_syms = list(set().union(*(df['sym'].unique() for df in dict_options.values())))
    all_syms = pd.Categorical([s for s in all_syms if pd.notna(s)])
    
    # Memory-efficient merge operation
    cols = ['date', 'symbol', 'bid', 'ask', 'mid', 'delta', 'spread']
    return options[cols].merge(pd.DataFrame({'symbol': all_syms}), on='symbol', how='inner')

# Signal State Machine Logic
def get_list_newsig_cycle(sigdata, days_to_set):
    sigdata_copy = sigdata.copy()
    sigdata_copy['sig'] = 0
    
    for i, d in enumerate(sigdata_copy.index):
        if i == 0: continue
        
        previous_sig = sigdata_copy.iloc[i-1]['sig']
        
        # State transition logic
        if previous_sig == 0:  # Currently out of market
            sig = 1 if sigdata_copy.at[d,'enter_conds'] else 0
        elif previous_sig == 1:  # Currently in market
            sig = 0 if sigdata_copy.at[d,'exit_conds'] else 1
            
        sigdata_copy.at[d,'sig'] = sig
